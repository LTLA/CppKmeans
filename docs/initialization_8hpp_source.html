45std::vector<INDEX_t> weighted_initialization(int ndim, INDEX_t nobs, const DATA_t* data, CLUSTER_t ncenters, ENGINE& eng) {
46 std::vector<DATA_t> mindist(nobs, 1);
47 std::vector<DATA_t> cumulative(nobs);
48 std::vector<INDEX_t> sofar;
49
50 if (!nobs) {
51 return sofar;
52 }
53 sofar.reserve(ncenters);
54
55 for (CLUSTER_t cen = 0; cen < ncenters; ++cen) {
56 INDEX_t counter = 0;
57 if (!sofar.empty()) {
58 auto last = sofar.back();
59
60 #pragma omp parallel for
61 for (INDEX_t obs = 0; obs < nobs; ++obs) {
62 if (mindist[obs]) {
63 const DATA_t* acopy = data + obs * ndim;
64 const DATA_t* scopy = data + last * ndim;
65 DATA_t r2 = 0;
66 for (int dim = 0; dim < ndim; ++dim, ++acopy, ++scopy) {
67 r2 += (*acopy - *scopy) * (*acopy - *scopy);
68 }
69
70 if (cen == 1 || r2 < mindist[obs]) {
71 mindist[obs] = r2;
72 }
73 }
74 }
75 } else {
76 counter = nobs;
77 }
78
79 cumulative[0] = mindist[0];
80 for (INDEX_t i = 1; i < nobs; ++i) {
81 cumulative[i] = cumulative[i-1] + mindist[i];
82 }
83
84 const auto total = cumulative.back();
85 if (total == 0) { // a.k.a. only duplicates left.
86 break;
87 }
88
89 INDEX_t chosen_id = 0;
90 do {
91 const DATA_t sampled_weight = total * aarand::standard_uniform(eng);
92 chosen_id = std::lower_bound(cumulative.begin(), cumulative.end(), sampled_weight) - cumulative.begin();
93
94 // We wrap this in a do/while to defend against edge cases where
95 // ties are chosen. The most obvious of these is when you get a
96 // `sampled_weight` of zero _and_ there exists a bunch of zeros at
97 // the start of `cumulative`. One could also get unexpected ties
98 // from limited precision in floating point comparisons, so we'll
99 // just be safe and implement a loop here, in the same vein as
100 // uniform01.
101 } while (chosen_id == nobs || mindist[chosen_id] == 0);
102
103 mindist[chosen_id] = 0;
104 sofar.push_back(chosen_id);
105 }
106
107 return sofar;
108}
