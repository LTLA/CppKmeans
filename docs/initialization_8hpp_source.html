21INDEX_t weighted_sample(const std::vector<DATA_t>& cumulative, const std::vector<DATA_t>& mindist, INDEX_t nobs, ENGINE& eng) {
22 auto total = cumulative.back();
23 INDEX_t chosen_id = 0;
24
25 do {
26 const DATA_t sampled_weight = total * aarand::standard_uniform(eng);
27 chosen_id = std::lower_bound(cumulative.begin(), cumulative.end(), sampled_weight) - cumulative.begin();
28
29 if (chosen_id == nobs) {
30 std::cout << chosen_id << "\t" << "YAY" << std::endl;
31 } else if (mindist[chosen_id] == 0) {
32 std::cout << chosen_id << "\t" << "YAY2" << std::endl;
33 }
34
35 // We wrap this in a do/while to defend against edge cases where
36 // ties are chosen. The most obvious of these is when you get a
37 // `sampled_weight` of zero _and_ there exists a bunch of zeros at
38 // the start of `cumulative`. One could also get unexpected ties
39 // from limited precision in floating point comparisons, so we'll
40 // just be safe and implement a loop here, in the same vein as
41 // uniform01.
42 } while (chosen_id == nobs || mindist[chosen_id] == 0);
43
44 return chosen_id;
45}

79std::vector<INDEX_t> weighted_initialization(int ndim, INDEX_t nobs, const DATA_t* data, CLUSTER_t ncenters, ENGINE& eng) {

81 std::vector<DATA_t> cumulative(nobs);

83
84 if (!nobs) {
85 return sofar;
86 }
87 sofar.reserve(ncenters);
88
89 for (CLUSTER_t cen = 0; cen < ncenters; ++cen) {
90 INDEX_t counter = 0;
91 if (!sofar.empty()) {

93
94 #pragma omp parallel for
95 for (INDEX_t obs = 0; obs < nobs; ++obs) {
96 if (mindist[obs]) {
97 const DATA_t* acopy = data + obs * ndim;
98 const DATA_t* scopy = data + last * ndim;
99 DATA_t r2 = 0;
100 for (int dim = 0; dim < ndim; ++dim, ++acopy, ++scopy) {
101 r2 += (*acopy - *scopy) * (*acopy - *scopy);
102 }
103
104 if (cen == 1 || r2 < mindist[obs]) {
105 mindist[obs] = r2;
106 }
107 }
108 }
109 } else {
110 counter = nobs;
111 }
112
113 cumulative[0] = mindist[0];
114 for (INDEX_t i = 1; i < nobs; ++i) {
115 cumulative[i] = cumulative[i-1] + mindist[i];
116 }
117
118 const auto total = cumulative.back();
119 if (total == 0) { // a.k.a. only duplicates left.
120 break;
121 }
122
123 auto chosen_id = weighted_sample(cumulative, mindist, nobs, eng);
124 mindist[chosen_id] = 0;
