46/*
47 * The class below represents 'ncp', which has a dual interpretation in the
48 * original Fortran implementation:
49 *
50 * - In the optimal-transfer stage, NCP(L) stores the step at which cluster L
51 * was last updated. Each step is just the observation index as the optimal
52 * transfer only does one pass through the dataset.
53 * - In the quick-transfer stage, NCP(L) stores the step at which cluster L is
54 * last updated plus M (i.e., the number of observations). Here, the step
55 * corresponding to an observation will be 'M * X + obs' for some integer X
56 * >= 0, where X is the iteration of the quick transfer.
57 *
58 * Note that these two definitions bleed into each other as the NCP(L) set by
59 * optimal_transfer is still being used in the first few iterations of
60 * quick_transfer before it eventually gets written. The easiest way to
61 * interpret this is to consider the optimal transfer as "iteration -1" from
62 * the perspective of the quick transfer iterations.
63 *
64 * In short, this data structure specifies whether a cluster was modified
65 * within the last M steps. This counts steps in both optimal_transfer and
66 * quick_transfer, and considers modifications from both calls.
67 */
68template<typename Index_>
69class UpdateHistory {
70public:
71 static constexpr int8_t max_quick_iterations = 50;
72
73private:
74 /*
75 * The problem with the original implementation is that the integers are
76 * expected to hold 'max_quick_iterations * M'. For a templated integer
77 * type, that might not be possible, so instead we split it into two
78 * vectors; one holds the last iteration at which the cluster was modified,
79 * the other holds the last observation used in the modification.
80 */
81 Index_ my_last_observation = 0;
82
83 // We use 8-bit ints to save some space, with signing for the special values.
84 int8_t my_last_iteration = init;
85
86 static constexpr int8_t init = -3;
87 static constexpr int8_t unchanged = -2;
88
89public:
90 void set_unchanged() {
91 my_last_observation = unchanged;
92 }
93
94 // We treat the optimal_transfer as "iteration -1" here.
95 void set_optimal(Index_ obs) {
96 my_last_iteration = -1;
97 my_last_observation = obs;
98 }
99
100 // Here, iter should be from '[0, max_quick_iterations)'.
101 void set_quick(int8_t iter, Index_ obs) {
102 my_last_iteration = iter;
103 my_last_observation = obs;
104 }
105
106public:
107 bool is_unchanged() const {
108 return my_last_iteration == unchanged;
109 }
110
111public:
112 bool changed_after(int8_t iter, Index_ obs) const {
113 if (my_last_iteration == iter) {
114 return my_last_observation > obs;
115 } else {
116 return my_last_iteration > iter;
117 }
118 }
119
120 bool changed_after_or_at(int8_t iter, Index_ obs) const {
121 if (my_last_iteration == iter) {
122 return my_last_observation >= obs;
123 } else {
124 return my_last_iteration > iter;
125 }
126 }
127};

129/*
130 * The class below represents 'live', which has a tricky interpretation.
131 *
132 * - Before each optimal transfer call, LIVE(L) stores the observation at which
133 * cluster L was updated in the _previous_ call.
134 * - During the optimal transfer call, LIVE(L) is updated to the observation at
135 * which L was updated in this call, plus M (i.e., number of observations).
136 * - After the optimal transfer call, LIVE(L) is updated by subtracting M, so
137 * that the interpretation is correct in the next call.
138 *
139 * It basically tells us whether there was a recent transfer (optimal or quick)
140 * within the last M steps of optimal_transfer. If so, the cluster is "live".
141 */
142template<typename Index_>
143class LiveStatus {
144private:
145 enum class Event : uint8_t { NONE, PAST_OPT, CURRENT_OPT, QUICK, INIT };
146
147 /* The problem with the original implementation is that LIVE(L) needs to
148 * store at least 2*M, which might cause overflows in Index_. To avoid
149 * this, we split this information into two vectors:
150 *
151 * - 'my_had_recent_transfer' specifies specifies whether a transfer
152 * occurred in the current optimal_transfer call, or in the immediately
153 * preceding quick_transfer call. If this > PAST_OPT, the cluster is
154 * definitely live; if it is == PAST_OPT, it may or may not be live.
155 * - 'my_last_optimal_transfer' has two interpretations:
156 * - If 'my_had_recent_transfer == PAST_OPT', it specifies the
157 * observation at which the last transfer occurred in previous
158 * optimal_transfer call. If this is greater than the current
159 * observation, the cluster is live.
160 * - If 'my_had_recent_transfer == CURRENT_OPT', it specifies the
161 * observation at which the last transfer occurred in the current
162 * optimal_transfer call.
163 * - Otherwise it is undefined and should not be used.
164 *
165 * One might think that 'LiveStatus::my_last_optimal_transfer' is redundant
166 * with 'UpdateHistory::my_last_observation', but the former only tracks
167 * optimal transfers while the latter includes quick transfers.
168 */
169 Event my_had_recent_transfer = Event::INIT;
170 Index_ my_last_optimal_transfer = 0;
171
172public:
173 bool is_live(Index_ obs) const {
174 if (my_had_recent_transfer == Event::PAST_OPT) {
175 return my_last_optimal_transfer > obs;
176 } else {
177 return my_had_recent_transfer > Event::PAST_OPT;
178 }
179 }
180
181 void mark_current(Index_ obs) {
182 my_had_recent_transfer = Event::CURRENT_OPT;
183 my_last_optimal_transfer = obs;
184 }
185
186 void reset(bool was_quick_transferred) {
187 if (was_quick_transferred) {
188 my_had_recent_transfer = Event::QUICK;
189 } else if (my_had_recent_transfer == Event::CURRENT_OPT) {
190 my_had_recent_transfer = Event::PAST_OPT;
191 } else {
192 my_had_recent_transfer = Event::NONE;
193 }
194 }
195};

229template<typename Data_, typename Float_, typename Dim_>
230Float_ squared_distance_from_cluster(const Data_* data, const Float_* center, Dim_ ndim) {
231 Float_ output = 0;
232 for (decltype(ndim) dim = 0; dim < ndim; ++dim, ++data, ++center) {
233 Float_ delta = static_cast<Float_>(*data) - *center; // cast to float for consistent precision regardless of Data_.
234 output += delta * delta;
235 }
236 return output;

265template<typename Dim_, typename Data_, typename Index_, typename Cluster_, typename Float_>
266void transfer_point(Dim_ ndim, const Data_* obs_ptr, Index_ obs_id, Cluster_ l1, Cluster_ l2, Float_* centers, Cluster_* best_cluster, Workspace<Float_, Index_, Cluster_>& work) {
267 // Yes, casts to float are deliberate here, so that the
268 // multipliers can be computed correctly.
269 Float_ al1 = work.cluster_sizes[l1], alw = al1 - 1;
270 Float_ al2 = work.cluster_sizes[l2], alt = al2 + 1;
271
272 size_t long_ndim = ndim;
273 auto copy1 = centers + static_cast<size_t>(l1) * long_ndim; // cast to avoid overflow.
274 auto copy2 = centers + static_cast<size_t>(l2) * long_ndim;
275 for (decltype(ndim) dim = 0; dim < ndim; ++dim, ++copy1, ++copy2, ++obs_ptr) {
276 Float_ oval = *obs_ptr; // cast to float for consistent precision regardless of Data_.
277 *copy1 = (*copy1 * al1 - oval) / alw;
278 *copy2 = (*copy2 * al2 + oval) / alt;
279 }
280
281 --work.cluster_sizes[l1];
282 ++work.cluster_sizes[l2];
283
284 work.gain_multiplier[l1] = alw / al1;
285 work.loss_multiplier[l1] = (alw > 1 ? alw / (alw - 1) : big_number<Float_>());
286 work.loss_multiplier[l2] = alt / al2;
287 work.gain_multiplier[l2] = alt / (alt + 1);
288
289 best_cluster[obs_id] = l2;
290 work.second_best_cluster[obs_id] = l1;
291}

293/* ALGORITHM AS 136.1 APPL. STATIST. (1979) VOL.28, NO.1
294 * This is the OPtimal TRAnsfer stage.
295 * ----------------------
296 * Each point is re-assigned, if necessary, to the cluster that will induce a
297 * maximum reduction in the within-cluster sum of squares. In this stage,
298 * there is only one pass through the data.
299 */
300template<class Matrix_, typename Cluster_, typename Float_>
301bool optimal_transfer(const Matrix_& data, Workspace<Float_, typename Matrix_::index_type, Cluster_>& work, Cluster_ ncenters, Float_* centers, Cluster_* best_cluster) {
302 auto nobs = data.num_observations();
303 auto ndim = data.num_dimensions();
304 auto matwork = data.create_workspace();
305 size_t long_ndim = ndim;
306
307 for (decltype(nobs) obs = 0; obs < nobs; ++obs) {
308 ++work.optra_steps_since_last_transfer;
309
310 auto l1 = best_cluster[obs];
311 if (work.cluster_sizes[l1] != 1) {
312 auto obs_ptr = data.get_observation(obs, matwork);
313
314 // Need to update the WCSS loss if this is (i) the first call to
315 // optimal_transfer, or (ii) if the cluster center was updated
316 // earlier in the current optimal_transfer call. No need to worry
317 // about quick_transfer as all WCSS losses are guaranteed to be
318 // accurate when we exit from that function.
319 auto& wcss_loss = work.wcss_loss[obs];
320 auto& history1 = work.update_history[l1];
321 if (!history1.is_unchanged()) {
322 auto l1_ptr = centers + long_ndim * static_cast<size_t>(l1); // cast to avoid overflow.
323 wcss_loss = squared_distance_from_cluster(obs_ptr, l1_ptr, ndim) * work.loss_multiplier[l1];
324 }
325
326 // Find the cluster with minimum WCSS gain.
327 auto l2 = work.second_best_cluster[obs];
328 auto original_l2 = l2;
329 auto l2_ptr = centers + long_ndim * static_cast<size_t>(l2); // cast to avoid overflow.
330 auto wcss_gain = squared_distance_from_cluster(obs_ptr, l2_ptr, ndim) * work.gain_multiplier[l2];
331
332 auto check_best_cluster = [&](Cluster_ cen) {
333 auto cen_ptr = centers + long_ndim * static_cast<size_t>(cen); // cast to avoid overflow.
334 auto candidate = squared_distance_from_cluster(obs_ptr, cen_ptr, ndim) * work.gain_multiplier[cen];
335 if (candidate < wcss_gain) {
336 wcss_gain = candidate;
337 l2 = cen;
338 }
339 };
340
341 // If the best cluster is live, we need to consider all other clusters.
342 // Otherwise, we only need to consider other live clusters for transfer.
343 auto& live1 = work.live_set[l1];
344 if (live1.is_live(obs)) {
345 for (Cluster_ cen = 0; cen < ncenters; ++cen) {
346 if (cen != l1 && cen != original_l2) {
347 check_best_cluster(cen);
348 }
349 }
350 } else {
351 for (Cluster_ cen = 0; cen < ncenters; ++cen) {
352 if (cen != l1 && cen != original_l2 && work.live_set[cen].is_live(obs)) {
353 check_best_cluster(cen);
354 }
355 }
356 }
357
358 // Deciding whether to make the transfer based on the change to the WCSS.
359 if (wcss_gain >= wcss_loss) {
360 work.second_best_cluster[obs] = l2;
361 } else {
362 work.optra_steps_since_last_transfer = 0;
363
364 live1.mark_current(obs);
365 work.live_set[l2].mark_current(obs);
366 history1.set_optimal(obs);
367 work.update_history[l2].set_optimal(obs);
368
369 transfer_point(ndim, obs_ptr, obs, l1, l2, centers, best_cluster, work);
370 }
371 }
372
373 // Stop if we've iterated through the entire dataset and no transfer of
374 // any kind took place, be it optimal or quick.
375 if (work.optra_steps_since_last_transfer == nobs) {
376 return true;
377 }
378 }
379
380 return false;
381}

383/* ALGORITHM AS 136.2 APPL. STATIST. (1979) VOL.28, NO.1
384 * This is the Quick TRANsfer stage.
385 * --------------------
386 * IC1(I) is the cluster which point I currently belongs to.
387 * IC2(I) is the cluster which point I is most likely to be transferred to.
388 *
389 * For each point I, IC1(I) & IC2(I) are switched, if necessary, to reduce
390 * within-cluster sum of squares. The cluster centres are updated after each
391 * step. In this stage, we loop through the data until no further change is to
392 * take place, or we hit an iteration limit, whichever is first.
393 */
394template<class Matrix_, typename Cluster_, typename Float_>
395std::pair<bool, bool> quick_transfer(const Matrix_& data, Workspace<Float_, typename Matrix_::index_type, Cluster_>& work, Float_* centers, Cluster_* best_cluster) {
396 bool had_transfer = false;
397 std::fill(work.was_quick_transferred.begin(), work.was_quick_transferred.end(), 0);
398
399 auto nobs = data.num_observations();
400 auto matwork = data.create_workspace();
401 auto ndim = data.num_dimensions();
402 size_t long_ndim = data.num_dimensions();
403
404 typedef decltype(nobs) Index_;
405 Index_ steps_since_last_quick_transfer = 0;
406
407 for (int8_t it = 0; it < UpdateHistory<Index_>::max_quick_iterations; ++it) {
408 int8_t prev_it = it - 1;
409
410 for (decltype(nobs) obs = 0; obs < nobs; ++obs) {
411 ++steps_since_last_quick_transfer;
412 auto l1 = best_cluster[obs];
413
414 if (work.cluster_sizes[l1] != 1) {
415 const typename Matrix_::data_type* obs_ptr = NULL;
416
417 // Need to update the WCSS loss if the cluster was updated recently.
418 // Otherwise, we must have already updated the WCSS in a previous
419 // iteration of the outermost loop, so this can be skipped.
420 //
421 // Note that we use changed_at_or_after; if the same
422 // observation was changed in the previous iteration of the
423 // outermost loop, its WCSS loss won't have been updated yet.
424 auto& history1 = work.update_history[l1];
425 if (history1.changed_after_or_at(prev_it, obs)) {
426 auto l1_ptr = centers + static_cast<size_t>(l1) * long_ndim; // cast to avoid overflow.
427 obs_ptr = data.get_observation(obs, matwork);
428 work.wcss_loss[obs] = squared_distance_from_cluster(obs_ptr, l1_ptr, ndim) * work.loss_multiplier[l1];
429 }
430
431 // If neither the best or second-best clusters have changed
432 // after the previous iteration that we visited this
433 // observation, then there's no point reevaluating the
434 // transfer, because nothing's going to be different anyway.
435 auto l2 = work.second_best_cluster[obs];
436 auto& history2 = work.update_history[l2];
437 if (history1.changed_after(prev_it, obs) || history2.changed_after(prev_it, obs)) {
438 if (obs_ptr == NULL) {
439 obs_ptr = data.get_observation(obs, matwork);
440 }
441 auto l2_ptr = centers + static_cast<size_t>(l2) * long_ndim; // cast to avoid overflow.
442 auto wcss_gain = squared_distance_from_cluster(obs_ptr, l2_ptr, ndim) * work.gain_multiplier[l2];
443
444 if (wcss_gain < work.wcss_loss[obs]) {
445 had_transfer = true;
446 steps_since_last_quick_transfer = 0;
447
448 work.was_quick_transferred[l1] = true;
449 work.was_quick_transferred[l2] = true;
450
451 history1.set_quick(it, obs);
452 history2.set_quick(it, obs);
453
454 transfer_point(ndim, obs_ptr, obs, l1, l2, centers, best_cluster, work);
455 }
456 }
457 }
458
459 if (steps_since_last_quick_transfer == nobs) {
460 // Quit early if no transfer occurred within the past 'nobs'
461 // steps, as we've already converged for each observation.
462 return std::make_pair(had_transfer, false);
463 }
464 }
465 }
466
467 return std::make_pair(had_transfer, true);
468}

504template<typename Matrix_ = SimpleMatrix<double, int>, typename Cluster_ = int, typename Float_ = double>
505class RefineHartiganWong : public Refine<Matrix_, Cluster_, Float_> {
506public:
510 RefineHartiganWong(RefineHartiganWongOptions options) : my_options(std::move(options)) {}
511
515 RefineHartiganWong() = default;
516
517private:
518 RefineHartiganWongOptions my_options;
519 typedef typename Matrix_::index_type Index_;
520
521public:
526 RefineHartiganWongOptions& get_options() {
527 return my_options;
528 }
529
530public:
531 Details<Index_> run(const Matrix_& data, Cluster_ ncenters, Float_* centers, Cluster_* clusters) const {
532 auto nobs = data.num_observations();
533 if (internal::is_edge_case(nobs, ncenters)) {
534 return internal::process_edge_case(data, ncenters, centers, clusters);
535 }
536
537 RefineHartiganWong_internal::Workspace<Float_, Index_, Cluster_> work(nobs, ncenters);
538
539 RefineHartiganWong_internal::find_closest_two_centers(data, ncenters, centers, clusters, work.second_best_cluster, my_options.num_threads);
540 for (Index_ obs = 0; obs < nobs; ++obs) {
541 ++work.cluster_sizes[clusters[obs]];
542 }
543 internal::compute_centroids(data, ncenters, centers, clusters, work.cluster_sizes);
544
545 for (Cluster_ cen = 0; cen < ncenters; ++cen) {
546 Float_ num = work.cluster_sizes[cen]; // yes, cast is deliberate here so that the multipliers can be computed correctly.
547 work.gain_multiplier[cen] = num / (num + 1);
548 work.loss_multiplier[cen] = (num > 1 ? num / (num - 1) : RefineHartiganWong_internal::big_number<Float_>());
549 }
550
551 int iter = 0;
552 int ifault = 0;
553 while ((++iter) <= my_options.max_iterations) {
554 bool finished = RefineHartiganWong_internal::optimal_transfer(data, work, ncenters, centers, clusters);
555 if (finished) {
556 break;
557 }
558
559 auto quick_status = RefineHartiganWong_internal::quick_transfer(data, work, centers, clusters);
560 if (quick_status.second) { // Hit the quick transfer iteration limit.
561 ifault = 4;
562 break;
563 }
564 if (quick_status.first) { // At least one transfer was performed.
565 work.optra_steps_since_last_transfer = 0;
566 }
567
568 // If there are only two clusters, there is no need to re-enter the optimal transfer stage.
569 if (ncenters == 2) {
570 break;
571 }
572
573 // If we get to this point, there must have been no transfers in
574 // the final quick_transfer iteration. This implies that all WCSS
575 // losses are up to date, as quick_transfer updated everything for
576 // us; so we can set the status to 'unchanged' for all clusters.
577 for (auto& u : work.update_history) {
578 u.set_unchanged();
579 }
580
581 for (Cluster_ c = 0; c < ncenters; ++c) {
582 work.live_set[c].reset(work.was_quick_transferred[c]);
583 }
584 }
585
586 if (iter == my_options.max_iterations + 1) {
587 ifault = 2;
588 }
589
590 internal::compute_centroids(data, ncenters, centers, clusters, work.cluster_sizes);
591 return Details(std::move(work.cluster_sizes), iter, ifault);
592 }
593};
