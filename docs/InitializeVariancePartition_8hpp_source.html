35template<class Dim_, typename Value_, typename Float_>
36void compute_welford(Dim_ ndim, const Value_* dptr, Float_* center, Float_* dim_ss, Float_ count) {
37 for (Dim_ j = 0; j < ndim; ++j) {
38 Float_ val = dptr[j];
39 auto cur_center = center[j];
40 Float_ delta = val - cur_center;
41 Float_ new_center = cur_center + delta / count;
42 center[j] = new_center;
43 dim_ss[j] += (val - new_center) * delta;
44 }
45}

105 Cluster_ run(const Matrix_& data, Cluster_ ncenters, Float_* centers) const {
106 auto nobs = data.num_observations();
107 auto ndim = data.num_dimensions();
108 if (nobs == 0 || ndim == 0) {
109 return 0;
110 }
111
112 std::vector<std::vector<typename Matrix_::index_type> > assignments(ncenters);
113 assignments[0].resize(nobs);
114 std::iota(assignments.front().begin(), assignments.front().end(), 0);
115
116 std::vector<std::vector<Float_> > dim_ss(ncenters);
117 {
118 auto& cur_ss = dim_ss[0];
119 cur_ss.resize(ndim);
120 std::fill_n(centers, ndim, 0);

122 for (decltype(nobs) i = 0; i < nobs; ++i) {
123 auto dptr = data.get_observation(matwork);
124 InitializeVariancePartition_internal::compute_welford(ndim, dptr, centers, cur_ss.data(), static_cast<Float_>(i + 1));
125 }
126 }

128 std::priority_queue<std::pair<Float_, Cluster_> > highest;
129 auto add_to_queue = [&](Cluster_ i) {
130 const auto& cur_ss = dim_ss[i];
131 Float_ sum_ss = std::accumulate(cur_ss.begin(), cur_ss.end(), static_cast<Float_>(0));
132
133 // Instead of dividing by N and then remultiplying by pow(N, adjustment), we just
134 // divide by pow(N, 1 - adjustment) to save some time and precision.
135 sum_ss /= std::pow(assignments[i].size(), 1.0 - my_options.size_adjustment);
136
137 highest.emplace(sum_ss, i);
138 };
139 add_to_queue(0);
140
141 std::vector<typename Matrix_::index_type> cur_assignments_copy;
142 size_t long_ndim = ndim;
143
144 for (Cluster_ cluster = 1; cluster < ncenters; ++cluster) {
145 auto chosen = highest.top();
146 if (chosen.first == 0) {
147 return cluster; // no point continuing, we're at zero variance already.
148 }
149 highest.pop();
150
151 auto* cur_center = centers + static_cast<size_t>(chosen.second) * long_ndim; // cast to size_t to avoid overflow issues.
152 auto& cur_ss = dim_ss[chosen.second];
153 auto& cur_assignments = assignments[chosen.second];
154
155 size_t top_dim = std::max_element(cur_ss.begin(), cur_ss.end()) - cur_ss.begin();
156 auto top_center = cur_center[top_dim];
157 std::fill_n(cur_center, ndim, 0);
158 std::fill(cur_ss.begin(), cur_ss.end(), 0);
159
160 auto* next_center = centers + static_cast<size_t>(cluster) * long_ndim; // again, size_t to avoid overflow.
161 std::fill_n(next_center, ndim, 0);
162 auto& next_ss = dim_ss[cluster];
163 next_ss.resize(ndim);
164 auto& next_assignments = assignments[cluster];
165
166 size_t num_in_cluster = cur_assignments.size();
167 auto work = data.create_workspace(cur_assignments.data(), num_in_cluster);
168 cur_assignments_copy.clear();
169
170 for (auto i : cur_assignments) {
171 auto dptr = data.get_observation(work);
172 if (dptr[top_dim] < top_center) {
173 cur_assignments_copy.push_back(i);
174 InitializeVariancePartition_internal::compute_welford(ndim, dptr, cur_center, cur_ss.data(), static_cast<Float_>(cur_assignments_copy.size()));
175 } else {
176 next_assignments.push_back(i);
177 InitializeVariancePartition_internal::compute_welford(ndim, dptr, next_center, next_ss.data(), static_cast<Float_>(next_assignments.size()));
178 }
179 }
180
181 // If one or the other is empty, then this entire procedure short
182 // circuits as all future iterations will just re-select this
183 // cluster (which won't get partitioned properly anyway). In the
184 // bigger picture, the quick exit out of the iterations is correct
185 // as we should only fail to partition in this manner if all points
186 // within each remaining cluster are identical.
187 if (next_assignments.empty() || cur_assignments_copy.empty()) {
188 return cluster;
189 }
190
191 cur_assignments.swap(cur_assignments_copy);
192 add_to_queue(chosen.second);
193 add_to_queue(cluster);
194 }
195
196 return ncenters;
197 }
