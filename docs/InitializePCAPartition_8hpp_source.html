131 static DATA_t normalize(int ndim, DATA_t* x) {
132 DATA_t ss = 0;
133 for (int d = 0; d < ndim; ++d) {
134 ss += x[d] * x[d];
135 }

137 if (ss) {
138 ss = std::sqrt(ss);
139 for (int d = 0; d < ndim; ++d) {
140 x[d] /= ss;

147 std::vector<DATA_t> compute_pc1(int ndim, const std::vector<INDEX_t>& chosen, const DATA_t* data, const DATA_t* center, Rng& eng) const {
148 std::vector<DATA_t> delta(ndim);
149 std::vector<DATA_t> cov(ndim * ndim);
150
151 // Computing the lower triangle of the covariance matrix.
152 for (auto i : chosen) {
153 auto dptr = data + i * ndim;
154 for (int j = 0; j < ndim; ++j) {
155 delta[j] = dptr[j] - center[j];
156 }
157 for (int j = 0; j < ndim; ++j) {
158 for (int k = 0; k <= j; ++k) {
159 cov[j * ndim + k] += delta[j] * delta[k];
160 }
161 }
162 }
163
164 // Filling in the other side of the matrix, to enable cache-efficient multiplication.
165 for (size_t j = 0; j < ndim; ++j) {
166 for (size_t k = j + 1; k < ndim; ++k) {
167 cov[j * ndim + k] = cov[k * ndim + j];
168 }
169 }
170
171 // Defining a random starting vector.
172 std::vector<DATA_t> output(ndim);
173 while (1) {
174 for (int d = 0; d < ndim - 1; d += 2) {
175 auto sampled = aarand::standard_normal<DATA_t>(eng);
176 output[d] = sampled.first;
177 output[d + 1] = sampled.second;
178 }
179 if (ndim % 2) {
180 output.back() = aarand::standard_normal<DATA_t>(eng).first;
181 }
182 if (normalize(ndim, output.data())) {
183 break;
184 }
185 }
186
187 // Applying power iterations.
188 for (int i = 0; i < iters; ++i) {
189 for (size_t j = 0; j < ndim; ++j) {
190 // As the matrix is symmetric, we can use inner_product.
191 // This technically computes the transpose of the matrix
192 // with the vector, but it's all the same, so whatever.
193 delta[j] = std::inner_product(output.begin(), output.end(), cov.data() + j * ndim, static_cast<DATA_t>(0.0));
194 }
195
196 // Normalizing the matrix.
197 auto l2 = normalize(ndim, delta.data());
198
199 // We want to know if SIGMA * output = lambda * output, i.e., l2 * delta = lambda * output.
200 // If we use l2 as a working estimate for lambda, we're basically just testing the difference
201 // between delta and output. We compute the error and compare this to the tolerance.
202 DATA_t err = 0;
203 for (int d = 0; d < ndim; ++d) {
204 DATA_t diff = delta[d] - output[i];
205 err += diff * diff;
206 }
207 if (std::sqrt(err) < tol) {
208 break;
209 }
210
211 std::copy(delta.begin(), delta.end(), output.begin());
212 }
213
214 return output;
215 }

243 static DATA_t update_mrse(int ndim, const std::vector<INDEX_t>& chosen, const DATA_t* data, DATA_t* center) {
244 compute_center(ndim, chosen, data, center);
245
246 DATA_t curmrse = 0;
247 for (auto i : chosen) {
248 auto dptr = data + i * ndim;
249 for (int d = 0; d < ndim; ++d) {
250 DATA_t delta = dptr[d] - center[d];
251 curmrse += delta * delta;
252 }
253 }
254
255 return curmrse / chosen.size();

261 /*
262 * @param ndim Number of dimensions.
263 * @param nobs Number of observations.
264 * @param data Pointer to an array where the dimensions are rows and the observations are columns.
265 * Data should be stored in column-major format.
266 * @param ncenters Number of centers to pick.
267 * @param[out] centers Pointer to a `ndim`-by-`ncenters` array where columns are cluster centers and rows are dimensions.
268 * On output, this will contain the final centroid locations for each cluster.
269 * Data should be stored in column-major order.
270 * @param clusters Pointer to an array of length `nobs`.
271 * This is used as a buffer and the contents on output should not be used.
272 *
273 * @return `centers` is filled with the new cluster centers.
274 * The number of filled centers is returned, see `Initializer::run()`.
275 */
276 CLUSTER_t run(int ndim, INDEX_t nobs, const DATA_t* data, CLUSTER_t ncenters, DATA_t* centers, CLUSTER_t* clusters) {
277 if (nobs == 0) {
278 return 0;
279 }
280
281 std::mt19937_64 rng(seed);
282 std::vector<DATA_t> mrse(ncenters);
283 std::vector<std::vector<INDEX_t> > assignments(ncenters);
284
285 // Setting up the zero'th cluster. (No need to actually compute the
286 // MRSE at this point, as there's nothing to compare it to.)
287 compute_center(ndim, nobs, data, centers);
288 assignments[0].resize(nobs);
289 std::iota(assignments.front().begin(), assignments.front().end(), 0);
290 std::fill(clusters, clusters + nobs, 0);
291
292 for (CLUSTER_t cluster = 1; cluster < ncenters; ++cluster) {
293 DATA_t worst_ss = 0;
294 INDEX_t worst_cluster = 0;
295 for (CLUSTER_t i = 0; i < cluster; ++i) {
296 DATA_t multiplier = assignments[i].size();
297 if (observation_cap && static_cast<DATA_t>(observation_cap) < multiplier) {
298 multiplier = observation_cap;
299 }
300
301 DATA_t pseudo_ss = mrse[i] * multiplier;
302 if (pseudo_ss > worst_ss) {
303 worst_ss = pseudo_ss;
304 worst_cluster = i;
305 }
306 }
307
308 // Extracting the principal component for this bad boy.
309 auto worst_center = centers + worst_cluster * ndim;
310 auto& worst_assignments = assignments[worst_cluster];
311 auto pc1 = compute_pc1(ndim, worst_assignments, data, worst_center, rng);
312
313 // Projecting all points in this cluster along PC1. The center lies
314 // at zero, so everything positive (on one side of the hyperplane
315 // orthogonal to PC1 and passing through the center) gets bumped to
316 // the next cluster.
317 std::vector<INDEX_t>& new_assignments = assignments[cluster];
318 std::vector<INDEX_t> worst_assignments2;
319 double accumulated = 0;
320 for (auto i : worst_assignments) {
321 auto dptr = data + i * ndim;
322 DATA_t proj = 0;
323 for (int d = 0; d < ndim; ++d) {
324 proj += (dptr[d] - worst_center[d]) * pc1[d];
325 }
326
327 accumulated += proj;
328 if (proj > 0) {
329 new_assignments.push_back(i);
330 } else {
331 worst_assignments2.push_back(i);
332 }
333 }
334
335 // If one or the other is empty, then this entire procedure short
336 // circuits as all future iterations will just re-select this
337 // cluster (which won't get partitioned properly anyway). In the
338 // bigger picture, the quick exit out of the iterations is correct
339 // as we should only fail to partition in this manner if all points
340 // within each remaining cluster are identical.
341 if (new_assignments.empty() || worst_assignments2.empty()) {
342 return cluster;
343 }
344
345 for (auto i : new_assignments) {
346 clusters[i] = cluster;
347 }
348 worst_assignments.swap(worst_assignments2);
349
350 // Computing centers and MRSE.
351 auto new_center = centers + cluster * ndim;
352 mrse[cluster] = update_mrse(ndim, new_assignments, data, new_center);
353 mrse[worst_cluster] = update_mrse(ndim, worst_assignments, data, worst_center);
354 }
355
356 return ncenters;
357 }
