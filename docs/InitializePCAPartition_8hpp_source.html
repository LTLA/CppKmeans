126 static DATA_t normalize(int ndim, DATA_t* x) {
127 DATA_t ss = 0;
128 for (int d = 0; d < ndim; ++d) {

142 std::vector<DATA_t> compute_pc1(int ndim, const std::vector<INDEX_t>& chosen, const DATA_t* data, const DATA_t* center, Rng& eng) const {
143 std::vector<DATA_t> delta(ndim);
144 std::vector<DATA_t> cov(ndim * ndim);
145
146 // Computing the lower triangle of the covariance matrix.
147 for (auto i : chosen) {
148 auto dptr = data + i * ndim;
149 for (int j = 0; j < ndim; ++j) {
150 delta[j] = dptr[j] - center[j];
151 }
152 for (int j = 0; j < ndim; ++j) {
153 for (int k = 0; k <= j; ++k) {
154 cov[j * ndim + k] += delta[j] * delta[k];
155 }
156 }
157 }
158
159 // Filling in the other side of the matrix, to enable cache-efficient multiplication.
160 for (size_t j = 0; j < ndim; ++j) {
161 for (size_t k = j + 1; k < ndim; ++k) {
162 cov[j * ndim + k] = cov[k * ndim + j];
163 }
164 }
165
166 // Defining a random starting vector.
167 std::vector<DATA_t> output(ndim);
168 while (1) {
169 for (int d = 0; d < ndim - 1; d += 2) {
170 auto sampled = aarand::standard_normal<DATA_t>(eng);
171 output[d] = sampled.first;
172 output[d + 1] = sampled.second;
173 }
174 if (ndim % 2) {
175 output.back() = aarand::standard_normal<DATA_t>(eng).first;
176 }
177 if (normalize(ndim, output.data())) {
178 break;
179 }
180 }
181
182 // Applying power iterations.
183 for (int i = 0; i < iters; ++i) {
184 for (size_t j = 0; j < ndim; ++j) {
185 // As the matrix is symmetric, we can use inner_product.
186 // This technically computes the transpose of the matrix
187 // with the vector, but it's all the same, so whatever.
188 delta[j] = std::inner_product(output.begin(), output.end(), cov.data() + j * ndim, static_cast<DATA_t>(0.0));
189 }
190
191 // Normalizing the matrix.
192 auto l2 = normalize(ndim, delta.data());
193
194 // We want to know if SIGMA * output = lambda * output, i.e., l2 * delta = lambda * output.
195 // If we use l2 as a working estimate for lambda, we're basically just testing the difference
196 // between delta and output. We compute the error and compare this to the tolerance.
197 DATA_t err = 0;
198 for (int d = 0; d < ndim; ++d) {
199 DATA_t diff = delta[d] - output[i];
200 err += diff * diff;
201 }
202 if (std::sqrt(err) < tol) {
203 break;
204 }
205
206 std::copy(delta.begin(), delta.end(), output.begin());
207 }
208
209 return output;
210 }

212 static void compute_center(int ndim, INDEX_t nobs, const DATA_t* data, DATA_t* center) {
213 std::fill(center, center + ndim, 0);

217 center[d] += dptr[d];
218 }
219 }
220 for (int d = 0; d < ndim; ++d) {
221 center[d] /= nobs;
222 }

238 static DATA_t update_mrse(int ndim, const std::vector<INDEX_t>& chosen, const DATA_t* data, DATA_t* center) {
239 compute_center(ndim, chosen, data, center);
240
241 DATA_t curmrse = 0;
242 for (auto i : chosen) {
243 auto dptr = data + i * ndim;
244 for (int d = 0; d < ndim; ++d) {
245 DATA_t delta = dptr[d] - center[d];
246 curmrse += delta * delta;
247 }
248 }
249
250 return curmrse / chosen.size();

256 /*
257 * @param ndim Number of dimensions.
258 * @param nobs Number of observations.
259 * @param data Pointer to an array where the dimensions are rows and the observations are columns.
260 * Data should be stored in column-major format.
261 * @param ncenters Number of centers to pick.
262 * @param[out] centers Pointer to a `ndim`-by-`ncenters` array where columns are cluster centers and rows are dimensions.
263 * On output, this will contain the final centroid locations for each cluster.
264 * Data should be stored in column-major order.
265 * @param clusters Pointer to an array of length `nobs`.
266 * This is used as a buffer and the contents on output should not be used.
267 *
268 * @return `centers` is filled with the new cluster centers.
269 * The number of filled centers is returned, see `Initializer::run()`.
270 */
271 CLUSTER_t run(int ndim, INDEX_t nobs, const DATA_t* data, CLUSTER_t ncenters, DATA_t* centers, CLUSTER_t* clusters) {
272 if (nobs == 0) {
273 return 0;
274 }
275
276 std::mt19937_64 rng(seed);
277 std::vector<DATA_t> mrse(ncenters);
278 std::vector<std::vector<INDEX_t> > assignments(ncenters);
279
280 // Setting up the zero'th cluster. (No need to actually compute the
281 // MRSE at this point, as there's nothing to compare it to.)
282 compute_center(ndim, nobs, data, centers);
283 assignments[0].resize(nobs);
284 std::iota(assignments.front().begin(), assignments.front().end(), 0);
285 std::fill(clusters, clusters + nobs, 0);
286
287 for (CLUSTER_t cluster = 1; cluster < ncenters; ++cluster) {
288 DATA_t worst_ss = 0;
289 INDEX_t worst_cluster = 0;
290 for (CLUSTER_t i = 0; i < cluster; ++i) {
291 DATA_t multiplier = assignments[i].size();
292 if (adjust != 1) {
293 multiplier = std::pow(static_cast<DATA_t>(multiplier), adjust);
294 }
295
296 DATA_t pseudo_ss = mrse[i] * multiplier;
297 if (pseudo_ss > worst_ss) {
298 worst_ss = pseudo_ss;
299 worst_cluster = i;
300 }
301 }
302
303 // Extracting the principal component for this bad boy.
304 auto worst_center = centers + worst_cluster * ndim;
305 auto& worst_assignments = assignments[worst_cluster];
306 auto pc1 = compute_pc1(ndim, worst_assignments, data, worst_center, rng);
307
308 // Projecting all points in this cluster along PC1. The center lies
309 // at zero, so everything positive (on one side of the hyperplane
310 // orthogonal to PC1 and passing through the center) gets bumped to
311 // the next cluster.
312 std::vector<INDEX_t>& new_assignments = assignments[cluster];
313 std::vector<INDEX_t> worst_assignments2;
314 for (auto i : worst_assignments) {
315 auto dptr = data + i * ndim;
316 DATA_t proj = 0;
317 for (int d = 0; d < ndim; ++d) {
318 proj += (dptr[d] - worst_center[d]) * pc1[d];
319 }
320
321 if (proj > 0) {
322 new_assignments.push_back(i);
323 } else {
324 worst_assignments2.push_back(i);
325 }
326 }
327
328 // If one or the other is empty, then this entire procedure short
329 // circuits as all future iterations will just re-select this
330 // cluster (which won't get partitioned properly anyway). In the
331 // bigger picture, the quick exit out of the iterations is correct
332 // as we should only fail to partition in this manner if all points
333 // within each remaining cluster are identical.
334 if (new_assignments.empty() || worst_assignments2.empty()) {
335 return cluster;
336 }
337
338 for (auto i : new_assignments) {
339 clusters[i] = cluster;
340 }
341 worst_assignments.swap(worst_assignments2);
342
343 // Computing centers and MRSE.
344 auto new_center = centers + cluster * ndim;
345 mrse[cluster] = update_mrse(ndim, new_assignments, data, new_center);
346 mrse[worst_cluster] = update_mrse(ndim, worst_assignments, data, worst_center);
347 }
348
349 return ncenters;
350 }
