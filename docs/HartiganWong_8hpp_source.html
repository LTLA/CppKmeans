345  /* ALGORITHM AS 136.1 APPL. STATIST. (1979) VOL.28, NO.1
346  * This is the OPtimal TRAnsfer stage.
347  * ----------------------
348  * Each point is re-allocated, if necessary, to the cluster that
349  * will induce a maximum reduction in the within-cluster sum of
350  * squares.
351  */
352  void optimal_transfer(INDEX_t& indx) {
353  /* If cluster L is updated in the last quick-transfer stage, it
354  * belongs to the live set throughout this stage. Otherwise, at
355  * each step, it is not in the live set if it has not been updated
356  * in the last M optimal transfer steps. (AL: M being a synonym for
357  * the number of observations, here and in other functions.)
358  */
359  for (CLUSTER_t cen = 0; cen < num_centers; ++cen) {
360  if (itran[cen]) {
361  live[cen] = num_obs; // AL: using 0-index, so no need for +1.
362  }
363  }
364 
365  for (INDEX_t obs = 0; obs < num_obs; ++obs) {
366  ++indx;
367  auto l1 = ic1[obs];
368 
369  // If point I is the only member of cluster L1, no transfer.
370  if (nc[l1] != 1) {
371  // If L1 has not yet been updated in this stage, no need to re-compute D(I).
372  if (!unchanged_ncp(l1)) {
373  d[obs] = squared_distance_from_cluster(obs, l1) * an1[l1];
374  }
375 
376  // Find the cluster with minimum R2.
377  auto l2 = ic2[obs];
378  auto ll = l2;
379  DATA_t r2 = squared_distance_from_cluster(obs, l2) * an2[l2];
380 
381  for (CLUSTER_t cen = 0; cen < num_centers; ++cen) {
382  /* If I >= LIVE(L1), then L1 is not in the live set. If this is
383  * true, we only need to consider clusters that are in the live
384  * set for possible transfer of point I. Otherwise, we need to
385  * consider all possible clusters.
386  */
387  if (obs >= live[l1] && obs >= live[cen] || cen == l1 || cen == ll) {
388  continue;
389  }
390 
391  DATA_t rr = r2 / an2[cen];
392  DATA_t dc = squared_distance_from_cluster(obs, cen);
393  if (dc < rr) {
394  r2 = dc * an2[cen];
395  l2 = cen;
396  }
397  }
398 
399  if (r2 >= d[obs]) {
400  // If no transfer is necessary, L2 is the new IC2(I).
401  ic2[obs] = l2;
402 
403  } else {
404  /* Update cluster centres, LIVE, NCP, AN1 & AN2 for clusters L1 and
405  * L2, and update IC1(I) & IC2(I).
406  */
407  indx = 0;
408  live[l1] = num_obs + obs;
409  live[l2] = num_obs + obs;
410  set_ncp(l1, obs);
411  set_ncp(l2, obs);
412 
413  transfer_point(obs, l1, l2);
414  }
415  }
416 
417  if (indx == num_obs) {
418  return;
419  }
420  }
421 
422  for (CLUSTER_t cen = 0; cen < num_centers; ++cen) {
423  itran[cen] = false;
424 
425  // LIVE(L) has to be decreased by M before re-entering OPTRA.
426  // This means that if I >= LIVE(L1) in the next OPTRA call,
427  // the last update must be >= M steps ago, as we effectively
428  // 'lapped' the previous update for this cluster.
429  live[cen] -= num_obs;
430  }
431 
432  return;
433  }

436  /* ALGORITHM AS 136.2 APPL. STATIST. (1979) VOL.28, NO.1
437  * This is the Quick TRANsfer stage.
438  * --------------------
439  * IC1(I) is the cluster which point I belongs to.
440  * IC2(I) is the cluster which point I is most likely to be
441  * transferred to.
442  *
443  * For each point I, IC1(I) & IC2(I) are switched, if necessary, to
444  * reduce within-cluster sum of squares. The cluster centres are
445  * updated after each step.
446  */
447  void quick_transfer (INDEX_t& indx, INDEX_t& imaxqtr) {
448  INDEX_t icoun = 0;
449  INDEX_t istep = 0;
450 
451  while (1) {
452  for (INDEX_t obs = 0; obs < num_obs; ++obs) {
453  ++icoun;
454  auto l1 = ic1[obs];
455 
456  // point I is the only member of cluster L1, no transfer.
457  if (nc[l1] != 1) {
458 
459  /* NCP(L) is equal to the step at which cluster L is last updated plus M.
460  * (AL: M is the notation for the number of observations, a.k.a. 'num_obs').
461  *
462  * If ISTEP > NCP(L1), no need to re-compute distance from point I to
463  * cluster L1. Note that if cluster L1 is last updated exactly M
464  * steps ago, we still need to compute the distance from point I to
465  * cluster L1.
466  */
467  if (le_ncp(l1, istep)) {
468  d[obs] = squared_distance_from_cluster(obs, l1) * an1[l1];
469  }
470 
471  // If ISTEP >= both NCP(L1) & NCP(L2) there will be no transfer of point I at this step.
472  auto l2 = ic2[obs];
473  if (lt_ncp(l1, istep) || lt_ncp(l2, istep)) {
474  if (squared_distance_from_cluster(obs, l2) < d[obs] / an2[l2]) {
475  /* Update cluster centres, NCP, NC, ITRAN, AN1 & AN2 for clusters
476  * L1 & L2. Also update IC1(I) & IC2(I). Note that if any
477  * updating occurs in this stage, INDX is set back to 0.
478  */
479  icoun = 0;
480  indx = 0;
481 
482  itran[l1] = true;
483  itran[l2] = true;
484  set_ncp(l1, istep + num_obs);
485  set_ncp(l2, istep + num_obs);
486  transfer_point(obs, l1, l2);
487  }
488  }
489  }
490 
491  // If no re-allocation took place in the last M steps, return.
492  if (icoun == num_obs) {
493  return;
494  }
495 
496  // AL: incrementing ISTEP after checks against NCP(L1), to avoid off-by-one
497  // errors after switching to zero-indexing for the observations.
498  ++istep;
499  if (istep >= imaxqtr) {
500  imaxqtr = -1;
501  return;
502  }
503  }
504  }

508  void transfer_point(INDEX_t obs, CLUSTER_t l1, CLUSTER_t l2) {
509  const DATA_t al1 = nc[l1], alw = al1 - 1;
510  const DATA_t al2 = nc[l2], alt = al2 + 1;
511 
512  auto copy1 = centers_ptr + l1 * num_dim;
513  auto copy2 = centers_ptr + l2 * num_dim;
514  auto acopy = data_ptr + obs * num_dim;
515  for (int dim = 0; dim < num_dim; ++dim, ++copy1, ++copy2, ++acopy) {
516  *copy1 = (*copy1 * al1 - *acopy) / alw;
517  *copy2 = (*copy2 * al2 + *acopy) / alt;
518  }
519 
520  --nc[l1];
521  ++nc[l2];
522 
523  an2[l1] = alw / al1;
524  an1[l1] = (alw > 1 ? alw / (alw - 1) : big);
525  an1[l2] = alt / al2;
526  an2[l2] = alt / (alt + 1);
527 
528  ic1[obs] = l2;
529  ic2[obs] = l1;
