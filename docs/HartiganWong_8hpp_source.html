147 num_dim = ndim;
148 num_obs = nobs;
149 data_ptr = data;
150 num_centers = ncenters;
151 centers_ptr = centers;
152 ic1 = clusters;
153
154 // Sizes taken from the .Fortran() call in stats::kmeans().
155 ic2.resize(num_obs);
156 nc.resize(num_centers);
157 an1.resize(num_centers);
158 an2.resize(num_centers);
159 d.resize(num_obs);
160
161 /* ITRAN(L) = 1 if cluster L is updated in the quick-transfer stage,
162 * = 0 otherwise
163 * In the optimal-transfer stage, NCP(L) stores the step at which
164 * cluster L is last updated.
165 * In the quick-transfer stage, NCP(L) stores the step at which
166 * cluster L is last updated plus M.
167 */
168 ncp.resize(num_centers);
169 itran.resize(num_centers);
170 live.resize(num_centers);
171
172 if (is_edge_case(num_obs, num_centers)) {
173 return process_edge_case(num_dim, num_obs, data_ptr, num_centers, centers_ptr, ic1);
174 }

176 /* For each point I, find its two closest centres, IC1(I) and
177 * IC2(I). Assign it to IC1(I).
178 */
179#ifndef KMEANS_CUSTOM_PARALLEL
180#ifdef _OPENMP
181 #pragma omp parallel for num_threads(nthreads)
182#endif
183 for (INDEX_t obs = 0; obs < num_obs; ++obs) {
184#else
185 KMEANS_CUSTOM_PARALLEL(num_obs, [&](INDEX_t first, INDEX_t last) -> void {
186 for (INDEX_t obs = first; obs < last; ++obs) {
187#endif
188 auto& best = ic1[obs];
189 best = 0;
190 DATA_t best_dist = squared_distance_from_cluster(obs, best);
191
192 auto& second = ic2[obs];
193 second = 1;
194 DATA_t second_dist = squared_distance_from_cluster(obs, second);
195
196 if (best_dist > second_dist) {
197 std::swap(best, second);
198 std::swap(best_dist, second_dist);
199 }
200
201 for (CLUSTER_t cen = 2; cen < num_centers; ++cen) {
202 DATA_t candidate_dist = squared_distance_from_cluster(obs, cen);
203 if (candidate_dist < second_dist) {
204 second_dist = candidate_dist;
205 second = cen;
206 if (candidate_dist < best_dist) {
207 std::swap(best_dist, second_dist);
208 std::swap(best, second);
209 }
210 }
211 }
212#ifndef KMEANS_CUSTOM_PARALLEL
213 }
214#else
215 }
216 }, nthreads);
217#endif

219 /* Update cluster centres to be the average of points contained
220 * within them.
221 * NC(L) := #{units in cluster L}, L = 1..K
222 */
223 std::fill(nc.begin(), nc.end(), 0);
224 for (INDEX_t obs = 0; obs < num_obs; ++obs) {
225 ++nc[ic1[obs]];

231 for (CLUSTER_t cen = 0; cen < num_centers; ++cen) {

237 * AN1(L) = NC(L) / (NC(L) - 1)
238 * AN2(L) = NC(L) / (NC(L) + 1)

240 const DATA_t num = nc[cen];
241 an2[cen] = num / (num + 1);
242 an1[cen] = (num > 1 ? num / (num - 1) : big);
243 }

260 * Each point is re-allocated, if necessary, to the cluster that will
261 * induce the maximum reduction in within-cluster sum of squares.

271 * be re-allocated to the cluster to which it is most likely to be
272 * transferred, IC2(I), from its present cluster, IC1(I).
273 * Loop through the data until no further change is to take place.

319 DATA_t squared_distance_from_cluster(INDEX_t pt, CLUSTER_t clust) const {
320 const DATA_t* acopy = data_ptr + pt * num_dim;
321 const DATA_t* ccopy = centers_ptr + clust * num_dim;
322 DATA_t output = 0;
323 for (int dim = 0; dim < num_dim; ++dim, ++acopy, ++ccopy) {
324 output += (*acopy - *ccopy) * (*acopy - *ccopy);
325 }
326 return output;
327 }

329private:
330 /* ALGORITHM AS 136.1 APPL. STATIST. (1979) VOL.28, NO.1
331 * This is the OPtimal TRAnsfer stage.
332 * ----------------------
333 * Each point is re-allocated, if necessary, to the cluster that
334 * will induce a maximum reduction in the within-cluster sum of
335 * squares.
336 */
337 void optimal_transfer(INDEX_t& indx) {
338 /* If cluster L is updated in the last quick-transfer stage, it
339 * belongs to the live set throughout this stage. Otherwise, at
340 * each step, it is not in the live set if it has not been updated
341 * in the last M optimal transfer steps. (AL: M being a synonym for
342 * the number of observations, here and in other functions.)
343 */
344 for (CLUSTER_t cen = 0; cen < num_centers; ++cen) {
345 if (itran[cen]) {
346 live[cen] = num_obs; // AL: using 0-index, so no need for +1.
347 }
348 }
349
350 for (INDEX_t obs = 0; obs < num_obs; ++obs) {
351 ++indx;
352 auto l1 = ic1[obs];
353
354 // If point I is the only member of cluster L1, no transfer.
355 if (nc[l1] != 1) {
356 // If L1 has not yet been updated in this stage, no need to re-compute D(I).
357 if (!unchanged_ncp(l1)) {
358 d[obs] = squared_distance_from_cluster(obs, l1) * an1[l1];
359 }
360
361 // Find the cluster with minimum R2.
362 auto l2 = ic2[obs];
363 auto ll = l2;
364 DATA_t r2 = squared_distance_from_cluster(obs, l2) * an2[l2];
365
366 for (CLUSTER_t cen = 0; cen < num_centers; ++cen) {
367 /* If I >= LIVE(L1), then L1 is not in the live set. If this is
368 * true, we only need to consider clusters that are in the live
369 * set for possible transfer of point I. Otherwise, we need to
370 * consider all possible clusters.
371 *
372 * Translation note: the original Fortran code doesn't have parentheses
373 * around the first &&, but .AND. has greater precedence than .OR.,
374 * so it should be faithful to add those parentheses for clarity
375 * (and to avoid GCC warnings).
376 */
377 if ((obs >= live[l1] && obs >= live[cen]) || cen == l1 || cen == ll) {
378 continue;
379 }
380
381 DATA_t rr = r2 / an2[cen];
382 DATA_t dc = squared_distance_from_cluster(obs, cen);
383 if (dc < rr) {
384 r2 = dc * an2[cen];
385 l2 = cen;
386 }
387 }
388
389 if (r2 >= d[obs]) {
390 // If no transfer is necessary, L2 is the new IC2(I).
391 ic2[obs] = l2;
392
393 } else {
394 /* Update cluster centres, LIVE, NCP, AN1 & AN2 for clusters L1 and
395 * L2, and update IC1(I) & IC2(I).
396 */
397 indx = 0;
398 live[l1] = num_obs + obs;
399 live[l2] = num_obs + obs;
400 set_ncp(l1, obs);
401 set_ncp(l2, obs);
402
403 transfer_point(obs, l1, l2);
404 }
405 }
406
407 if (indx == num_obs) {
408 return;
409 }
410 }
411
412 for (CLUSTER_t cen = 0; cen < num_centers; ++cen) {
413 itran[cen] = false;
414
415 // LIVE(L) has to be decreased by M before re-entering OPTRA.
416 // This means that if I >= LIVE(L1) in the next OPTRA call,
417 // the last update must be >= M steps ago, as we effectively
418 // 'lapped' the previous update for this cluster.
419 live[cen] -= num_obs;
420 }
421
422 return;
423 }

426 /* ALGORITHM AS 136.2 APPL. STATIST. (1979) VOL.28, NO.1
427 * This is the Quick TRANsfer stage.
428 * --------------------
429 * IC1(I) is the cluster which point I belongs to.
430 * IC2(I) is the cluster which point I is most likely to be
431 * transferred to.
432 *
433 * For each point I, IC1(I) & IC2(I) are switched, if necessary, to
434 * reduce within-cluster sum of squares. The cluster centres are
435 * updated after each step.
436 */
437 void quick_transfer (INDEX_t& indx, INDEX_t& imaxqtr) {
438 INDEX_t icoun = 0;
439 INDEX_t istep = 0;
440
441 while (1) {
442 for (INDEX_t obs = 0; obs < num_obs; ++obs) {
443 ++icoun;
444 auto l1 = ic1[obs];
445
446 // point I is the only member of cluster L1, no transfer.
447 if (nc[l1] != 1) {
448
449 /* NCP(L) is equal to the step at which cluster L is last updated plus M.
450 * (AL: M is the notation for the number of observations, a.k.a. 'num_obs').
451 *
452 * If ISTEP > NCP(L1), no need to re-compute distance from point I to
453 * cluster L1. Note that if cluster L1 is last updated exactly M
454 * steps ago, we still need to compute the distance from point I to
455 * cluster L1.
456 */
457 if (le_ncp(l1, istep)) {
458 d[obs] = squared_distance_from_cluster(obs, l1) * an1[l1];
459 }
460
461 // If ISTEP >= both NCP(L1) & NCP(L2) there will be no transfer of point I at this step.
462 auto l2 = ic2[obs];
463 if (lt_ncp(l1, istep) || lt_ncp(l2, istep)) {
464 if (squared_distance_from_cluster(obs, l2) < d[obs] / an2[l2]) {
465 /* Update cluster centres, NCP, NC, ITRAN, AN1 & AN2 for clusters
466 * L1 & L2. Also update IC1(I) & IC2(I). Note that if any
467 * updating occurs in this stage, INDX is set back to 0.
468 */
469 icoun = 0;
470 indx = 0;
471
472 itran[l1] = true;
473 itran[l2] = true;
474 set_ncp(l1, istep + num_obs);
475 set_ncp(l2, istep + num_obs);
476 transfer_point(obs, l1, l2);
477 }
478 }
479 }
480
481 // If no re-allocation took place in the last M steps, return.
482 if (icoun == num_obs) {
483 return;
484 }
485
486 // AL: incrementing ISTEP after checks against NCP(L1), to avoid off-by-one
487 // errors after switching to zero-indexing for the observations.
488 ++istep;
489 if (istep >= imaxqtr) {
490 imaxqtr = -1;
491 return;
492 }
493 }
494 }
495 }

497private:
498 void transfer_point(INDEX_t obs, CLUSTER_t l1, CLUSTER_t l2) {
499 const DATA_t al1 = nc[l1], alw = al1 - 1;
500 const DATA_t al2 = nc[l2], alt = al2 + 1;
501
502 auto copy1 = centers_ptr + l1 * num_dim;
503 auto copy2 = centers_ptr + l2 * num_dim;
504 auto acopy = data_ptr + obs * num_dim;
505 for (int dim = 0; dim < num_dim; ++dim, ++copy1, ++copy2, ++acopy) {
506 *copy1 = (*copy1 * al1 - *acopy) / alw;
507 *copy2 = (*copy2 * al2 + *acopy) / alt;
508 }
509
510 --nc[l1];
511 ++nc[l2];
512
513 an2[l1] = alw / al1;
514 an1[l1] = (alw > 1 ? alw / (alw - 1) : big);
515 an1[l2] = alt / al2;
516 an2[l2] = alt / (alt + 1);
517
518 ic1[obs] = l2;
519 ic2[obs] = l1;
520 }
521};
