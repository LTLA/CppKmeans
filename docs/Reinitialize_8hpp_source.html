126 CLUSTER_t run(int ndim, INDEX_t nobs, const DATA_t* data, CLUSTER_t ncenters, DATA_t* centers, CLUSTER_t* clusters) {
127 std::vector<DATA_t> mindist(nobs);
128 std::vector<DATA_t> cumulative(nobs);
129
130 if (recompute_clusters) {
131 QuickSearch x(ndim, ncenters, centers);
132 #pragma omp parallel for
133 for (INDEX_t obs = 0; obs < nobs; ++obs) {
134 clusters[obs] = x.find(data + obs * ndim);
135 }
136 }
137
138 // Technically, we could get the distances from QuickSearch,
139 // but this gets passaged through a sqrt(), so we just recompute
140 // it to avoid problems with loss of numerical precision when
141 // computing WCSS (which may cause headaches with equality
142 // comparisons during the random sampling).
143 #pragma omp parallel for
144 for (INDEX_t obs = 0; obs < nobs; ++obs) {
145 const DATA_t* o = data + ndim * obs;
146 const DATA_t* c = centers + ndim * clusters[obs];
147 auto& current = mindist[obs];
148 for (int d = 0; d < ndim; ++d) {
149 double delta = o[d] - c[d];
150 current += delta * delta;
151 }
152 }
153
154 std::mt19937_64 eng(seed);
155 DATA_t wcss = std::accumulate(mindist.begin(), mindist.end(), 0.0);
156 std::vector<CLUSTER_t> clusters_erased(nobs);
157 std::vector<DATA_t> mindist_erased(nobs);
158
159 for (CLUSTER_t c = 0; c < ncenters; ++c) {
160 std::copy(mindist.begin(), mindist.end(), mindist_erased.begin());
161 std::copy(clusters, clusters + nobs, clusters_erased.begin());
162
163 // Erasing cluster 'c' and search on the rest. For c > 0, this erasure
164 // is implicitly handled by the replacement with the sampled point.
165 QuickSearch x(ndim, ncenters - 1, centers + ndim);
166 #pragma omp parallel for
167 for (INDEX_t obs = 0; obs < nobs; ++obs) {
168 if (clusters_erased[obs] == c) {
169 auto closest = x.find_with_distance(data + obs * ndim);
170 mindist_erased[obs] = closest.second * closest.second;
171
172 // There is a frameshift on the cluster identities due to our
173 // removal and iterative replacement of cluster 'c'. Thus, we
174 // need to adjust the cluster IDs to ensure that they do not
175 // conflict with the original cluster identities, given that
176 // we need the originals for the later updates.
177 if (closest.first < c) {
178 clusters_erased[obs] = closest.first;
179 } else {
180 clusters_erased[obs] = closest.first + 1;
181 }
182 }
183 }
184
185 cumulative[0] = mindist_erased[0];
186 for (INDEX_t i = 1; i < nobs; ++i) {
187 cumulative[i] = cumulative[i-1] + mindist_erased[i];
188 }
189
190 // Performing multiple iterations to choose a new cluster center,
191 // making sure that the WCSS is lower than what it was before we
192 // revoked the existing cluster center.
193 std::vector<DATA_t> mindist_copy(nobs);
194 std::vector<CLUSTER_t> clusters_copy(nobs);
195
196 const DATA_t* chosen_ptr = NULL;
197 for (int it = 0; it < iterations; ++it) {
198 auto candidate = weighted_sample(cumulative, mindist_erased, nobs, eng);
199 auto candidate_ptr = data + candidate * ndim;
200 std::copy(mindist_erased.begin(), mindist_erased.end(), mindist_copy.begin());
201 std::copy(clusters_erased.begin(), clusters_erased.end(), clusters_copy.begin());
202
203 // Updating the minimum distances and the closest clusters.
204 #pragma omp parallel for
205 for (INDEX_t obs = 0; obs < nobs; ++obs) {
206 const DATA_t* o = data + ndim * obs;
207 DATA_t candidate_dist = 0;
208 for (int d = 0; d < ndim; ++d) {
209 double delta = candidate_ptr[d] - o[d];
210 candidate_dist += delta * delta;
211 }
212 if (candidate_dist < mindist_copy[obs]) {
213 mindist_copy[obs] = candidate_dist;
214 clusters_copy[obs] = c;
215 }
216 }
217
218 double wcss_copy = std::accumulate(mindist_copy.begin(), mindist_copy.end(), 0.0);
219 if (wcss > wcss_copy) {
220 mindist.swap(mindist_copy);
221 std::copy(clusters_copy.begin(), clusters_copy.end(), clusters);
222 wcss = wcss_copy;
223 chosen_ptr = candidate_ptr;
224 break;
225 }
226 }
227
228 if (c + 1 < ncenters) {
229 // Implicit erasure happens here:
230 // - The start of 'centers' contains the center of 'c' (possibly
231 // swapped in from a previous iteration).
232 // - We swap it with the center of 'c + 1', so the buffer for
233 // 'c + 1' now contains the center of 'c' and the start of
234 // 'centers' now contains the center for 'c + 1'.
235 // - We optionally replace the buffer for 'c + 1' with the
236 // chosen point. Or not, if it wasn't good enough.
237 auto target_ptr = centers + (c + 1) * ndim;
238 for (int d = 0; d < ndim; ++d) {
239 std::swap(target_ptr[d], centers[d]);
240 }
241 if (chosen_ptr) {
242 std::copy(chosen_ptr, chosen_ptr + ndim, target_ptr);
243 }
244 } else {
245 // Undoing the frameshift on the cluster centers, given
246 // that we effectively shifted everything to the right
247 // to allow a copy-free QuickSearch.
248 std::vector<DATA_t> holding(centers, centers + ndim);
249 for (CLUSTER_t c2 = 0; c2 < ncenters - 1; ++c2) {
250 auto dest = centers + c2 * ndim;
251 auto src = dest + ndim;
252 std::copy(src, src + ndim, dest);
253 }
254
255 auto final_ptr = centers + c * ndim;
256 if (chosen_ptr) {
257 std::copy(chosen_ptr, chosen_ptr + ndim, final_ptr);
258 } else {
259 std::copy(holding.begin(), holding.end(), final_ptr);
260 }
261 }
262 }
263
264 return ncenters;
265 }
