44 template<typename DATA_t = double, typename INDEX_t = int, typename CLUSTER_t = int>
45 void run(int ndim, INDEX_t nobs, const DATA_t* data, CLUSTER_t ncenters, DATA_t* centers, CLUSTER_t* clusters) const {
46 std::vector<DATA_t> mindist(nobs);
47 std::vector<DATA_t> cumulative(nobs);
48
49 if (recompute_clusters) {
50 QuickSearch x(ndim, ncenters, centers);
51 #pragma omp parallel for
52 for (INDEX_t obs = 0; obs < nobs; ++obs) {
53 clusters[obs] = x.find(data + obs * ndim);
54 }
55 }
56
57 // Technically, we could get the distances from QuickSearch,
58 // but this gets passaged through a sqrt(), so we just recompute
59 // it to avoid problems with loss of numerical precision when
60 // computing WCSS (which may cause headaches with equality
61 // comparisons during the random sampling).
62 #pragma omp parallel for
63 for (INDEX_t obs = 0; obs < nobs; ++obs) {
64 const DATA_t* o = data + ndim * obs;
65 const DATA_t* c = centers + ndim * clusters[obs];
66 auto& current = mindist[obs];
67 for (int d = 0; d < ndim; ++d) {
68 double delta = o[d] - c[d];
69 current += delta * delta;
70 }
71 }
72
73 std::mt19937_64 eng(seed);
74 DATA_t wcss = std::accumulate(mindist.begin(), mindist.end(), 0.0);
75 std::vector<CLUSTER_t> clusters_erased(nobs);
76 std::vector<DATA_t> mindist_erased(nobs);
77
78 for (CLUSTER_t c = 0; c < ncenters; ++c) {
79 std::copy(mindist.begin(), mindist.end(), mindist_erased.begin());
80 std::copy(clusters, clusters + nobs, clusters_erased.begin());
81
82 // Erasing cluster 'c' and search on the rest. For c > 0, this erasure
83 // is implicitly handled by the replacement with the sampled point.
84 QuickSearch x(ndim, ncenters - 1, centers + ndim);
85 #pragma omp parallel for
86 for (INDEX_t obs = 0; obs < nobs; ++obs) {
87 if (clusters_erased[obs] == c) {
88 auto closest = x.find_with_distance(data + obs * ndim);
89 mindist_erased[obs] = closest.second * closest.second;
90
91 // There is a frameshift on the cluster identities due to our
92 // removal and iterative replacement of cluster 'c'. Thus, we
93 // need to adjust the cluster IDs to ensure that they do not
94 // conflict with the original cluster identities, given that
95 // we need the originals for the later updates.
96 if (closest.first < c) {
97 clusters_erased[obs] = closest.first;
98 } else {
99 clusters_erased[obs] = closest.first + 1;
100 }
101 }
102 }
103
104 cumulative[0] = mindist_erased[0];
105 for (INDEX_t i = 1; i < nobs; ++i) {
106 cumulative[i] = cumulative[i-1] + mindist_erased[i];
107 }
108
109 // Performing multiple iterations to choose a new cluster center,
110 // making sure that the WCSS is lower than what it was before we
111 // revoked the existing cluster center.
112 std::vector<DATA_t> mindist_copy(nobs);
113 std::vector<CLUSTER_t> clusters_copy(nobs);
114
115 const DATA_t* chosen_ptr = NULL;
116 for (int it = 0; it < iterations; ++it) {
117 auto candidate = weighted_sample(cumulative, mindist_erased, nobs, eng);
118 auto candidate_ptr = data + candidate * ndim;
119 std::copy(mindist_erased.begin(), mindist_erased.end(), mindist_copy.begin());
120 std::copy(clusters_erased.begin(), clusters_erased.end(), clusters_copy.begin());
121
122 // Updating the minimum distances and the closest clusters.
123 #pragma omp parallel for
124 for (INDEX_t obs = 0; obs < nobs; ++obs) {
125 const DATA_t* o = data + ndim * obs;
126 DATA_t candidate_dist = 0;
127 for (int d = 0; d < ndim; ++d) {
128 double delta = candidate_ptr[d] - o[d];
129 candidate_dist += delta * delta;
130 }
131 if (candidate_dist < mindist_copy[obs]) {
132 mindist_copy[obs] = candidate_dist;
133 clusters_copy[obs] = c;
134 }
135 }
136
137 double wcss_copy = std::accumulate(mindist_copy.begin(), mindist_copy.end(), 0.0);
138 if (wcss > wcss_copy) {
139 mindist.swap(mindist_copy);
140 std::copy(clusters_copy.begin(), clusters_copy.end(), clusters);
141 wcss = wcss_copy;
142 chosen_ptr = candidate_ptr;
143 break;
144 }
145 }
146
147 if (c + 1 < ncenters) {
148 // Implicit erasure happens here:
149 // - The start of 'centers' contains the center of 'c' (possibly
150 // swapped in from a previous iteration).
151 // - We swap it with the center of 'c + 1', so the buffer for
152 // 'c + 1' now contains the center of 'c' and the start of
153 // 'centers' now contains the center for 'c + 1'.
154 // - We optionally replace the buffer for 'c + 1' with the
155 // chosen point. Or not, if it wasn't good enough.
156 auto target_ptr = centers + (c + 1) * ndim;
157 for (int d = 0; d < ndim; ++d) {
158 std::swap(target_ptr[d], centers[d]);
159 }
160 if (chosen_ptr) {
161 std::copy(chosen_ptr, chosen_ptr + ndim, target_ptr);
162 }
163 } else {
164 // Undoing the frameshift on the cluster centers, given
165 // that we effectively shifted everything to the right
166 // to allow a copy-free QuickSearch.
167 std::vector<DATA_t> holding(centers, centers + ndim);
168 for (CLUSTER_t c2 = 0; c2 < ncenters - 1; ++c2) {
169 auto dest = centers + c2 * ndim;
170 auto src = dest + ndim;
171 std::copy(src, src + ndim, dest);
172 }
173
174 auto final_ptr = centers + c * ndim;
175 if (chosen_ptr) {
176 std::copy(chosen_ptr, chosen_ptr + ndim, final_ptr);
177 } else {
178 std::copy(holding.begin(), holding.end(), final_ptr);
179 }
180 }
181 }
182
183 return;
184 }
