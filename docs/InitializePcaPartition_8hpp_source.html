58template<class Matrix_, typename Float_, class Engine_>
59void compute_pc1(
60 const Matrix_& data,
61 const std::vector<typename Matrix_::index_type>& chosen,
62 const Float_* center,
63 Engine_& eng,
64 Workspace<Float_>& work,
65 const powerit::Options& power_opts)
66{
67 auto ndim = data.num_dimensions();
68 size_t long_ndim = ndim;
69
70 auto matwork = data.create_workspace(chosen.data(), chosen.size());
71 for (size_t i = 0, end = chosen.size(); i < end; ++i) {
72 auto dptr = data.get_observation(matwork);
73 for (decltype(ndim) j = 0; j < ndim; ++j) {
74 work.delta[j] = static_cast<Float_>(dptr[j]) - center[j]; // cast to ensure consistent precision regardless of Data_.
75 }
76
77 size_t offset = 0;
78 for (decltype(ndim) j = 0; j < ndim; ++j, offset += long_ndim) {
79 size_t copy_offset = offset;
80 for (decltype(ndim) k = 0; k <= j; ++k, ++copy_offset) {
81 work.cov[copy_offset] += work.delta[j] * work.delta[k];
82 }
83 }
84 }
85
86 // Filling in the other side of the matrix, to enable cache-efficient multiplication.
87 size_t src_offset = 0;
88 for (decltype(ndim) j = 0; j < ndim; ++j, src_offset += long_ndim) {
89 size_t dest_offset = j;
90 size_t src_offset_copy = src_offset;
91 for (decltype(ndim) k = 0; k <= j; ++k, ++src_offset_copy, dest_offset += long_ndim) {
92 work.cov[dest_offset] = work.cov[src_offset_copy];
93 }
94 }
95
96 powerit::compute(ndim, work.cov.data(), /* row_major = */ true, work.pc.data(), eng, power_opts);
97}

99template<class Matrix_, typename Float_>
100void compute_center(const Matrix_& data, const std::vector<typename Matrix_::index_type>& chosen, Float_* center) {
101 auto ndim = data.num_dimensions();
102 std::fill_n(center, ndim, 0);
103
104 auto work = data.create_workspace(chosen.data(), chosen.size());
105 for (size_t i = 0, end = chosen.size(); i < end; ++i) {
106 auto dptr = data.get_observation(work);
107 for (decltype(ndim) d = 0; d < ndim; ++d) {
108 center[d] += static_cast<Float_>(dptr[d]); // cast to ensure consistent precision regardless of Data_.
109 }
110 }
111
112 for (decltype(ndim) d = 0; d < ndim; ++d) {
113 center[d] /= chosen.size();
114 }
115}

117template<class Matrix_, typename Float_>
118Float_ update_center_and_mrse(const Matrix_& data, const std::vector<typename Matrix_::index_type>& chosen, Float_* center) {
119 compute_center(data, chosen, center);
120
121 auto ndim = data.num_dimensions();
122 Float_ mrse = 0;
123 auto work = data.create_workspace(chosen.data(), chosen.size());
124 for (size_t i = 0, end = chosen.size(); i < end; ++i) {
125 auto dptr = data.get_observation(work);
126 for (decltype(ndim) d = 0; d < ndim; ++d) {
127 Float_ delta = static_cast<Float_>(dptr[d]) - center[d]; // cast to ensure consistent precision regardless of Data_.
128 mrse += delta * delta;
129 }
130 }
131
132 return mrse / chosen.size();
133}

173template<typename Matrix_ = SimpleMatrix<double, int>, typename Cluster_ = int, typename Float_ = double>
174class InitializePcaPartition : public Initialize<Matrix_, Cluster_, Float_> {
175public:
179 InitializePcaPartition(InitializePcaPartitionOptions options) : my_options(std::move(options)) {}
180
184 InitializePcaPartition() = default;
185
186private:
187 InitializePcaPartitionOptions my_options;
188
189public:
194 InitializePcaPartitionOptions& get_options() {
195 return my_options;
196 }
197
198public:
199 Cluster_ run(const Matrix_& data, Cluster_ ncenters, Float_* centers) const {
200 auto nobs = data.num_observations();
201 if (nobs == 0) {
202 return 0;
203 }
204
205 std::mt19937_64 rng(my_options.seed);
206 std::priority_queue<std::pair<Float_, Cluster_> > mrse;
207 std::vector<std::vector<typename Matrix_::index_type> > assignments(ncenters);
208
209 auto ndim = data.num_dimensions();
210 InitializePcaPartition_internal::Workspace<Float_> power_work(ndim);
211
212 // Setting up the zero'th cluster. (No need to actually compute the
213 // MRSE at this point, as there's nothing to compare it to.)
214 internal::compute_centroid(data, centers);
215 assignments[0].resize(nobs);
216 std::iota(assignments.front().begin(), assignments.front().end(), 0);
217 std::vector<typename Matrix_::index_type> replace_assignments;
218
219 for (Cluster_ cluster = 1; cluster < ncenters; ++cluster) {
220 Cluster_ worst_cluster = 0;
221 if (mrse.size()) {
222 worst_cluster = mrse.top().second;
223 mrse.pop();
224 }
225
226 // Extracting the principal component for this bad boy.
227 auto worst_center = centers + static_cast<size_t>(worst_cluster) * static_cast<size_t>(ndim); // cast to avoid overflow.
228 auto& worst_assignments = assignments[worst_cluster];
229 InitializePcaPartition_internal::compute_pc1(data, worst_assignments, worst_center, rng, power_work, my_options.power_iteration_options);
230 const auto& pc1 = power_work.pc;
231
232 // Projecting all points in this cluster along PC1. The center lies
233 // at zero, so everything positive (on one side of the hyperplane
234 // orthogonal to PC1 and passing through the center) gets bumped to
235 // the next cluster.
236 auto& new_assignments = assignments[cluster];
237 replace_assignments.clear();
238
239 size_t num_in_cluster = worst_assignments.size();
240 auto work = data.create_workspace(worst_assignments.data(), num_in_cluster);
241 for (auto i : worst_assignments) {
242 auto dptr = data.get_observation(work);
243
244 Float_ proj = 0;
245 for (decltype(ndim) d = 0; d < ndim; ++d, ++dptr) {
246 proj += (static_cast<Float_>(*dptr) - worst_center[d]) * pc1[d]; // cast for consistent precision regardless of Data_.
247 }
248
249 if (proj > 0) {
250 new_assignments.push_back(i);
251 } else {
252 replace_assignments.push_back(i);
253 }
254 }
255
256 // If one or the other is empty, then this entire procedure short
257 // circuits as all future iterations will just re-select this
258 // cluster (which won't get partitioned properly anyway). In the
259 // bigger picture, the quick exit out of the iterations is correct
260 // as we should only fail to partition in this manner if all points
261 // within each remaining cluster are identical.
262 if (new_assignments.empty() || replace_assignments.empty()) {
263 return cluster;
264 }
265
266 // Computing centers and MRSE.
267 auto new_center = centers + static_cast<size_t>(cluster) * static_cast<size_t>(ndim); // cast to avoid overflow.
268 auto new_mrse = InitializePcaPartition_internal::update_center_and_mrse(data, new_assignments, new_center);
269 mrse.emplace(new_mrse, cluster);
270
271 auto replace_mrse = InitializePcaPartition_internal::update_center_and_mrse(data, replace_assignments, worst_center);
272 mrse.emplace(replace_mrse, worst_cluster);
273
274 worst_assignments.swap(replace_assignments);
275 }
276
277 return ncenters;
278 }
